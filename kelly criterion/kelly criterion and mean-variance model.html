def simulator(rw = 1.0, rl = 1.0, p = 0.51):
    w = [1] * 5
    w_list = [[],[],[],[],[]]
    f = round((p*rw - (1 - p)*rw)/(rw*rl),3)
    F = [f, f*0.5,f*1.5,f*2,f*3]
    for i in range(10000):
        r = random.uniform(0,1)
        for j in range(len(F)):
            if r <= p:
                w[j] = w[j] * (1 + F[j] * rw)
            else:
                w[j] = w[j] * (1 - F[j] * rl)
            w_list[j].append(w[j])
    plt.plot(w_list[0], 'r', label = str(F[0]))
    plt.plot(w_list[1], 'b', label = str(F[1]))
    plt.plot(w_list[2], 'b', label = str(F[2]))
    plt.plot(w_list[3], 'b', label = str(F[3]))
    plt.plot(w_list[4], 'b', label = str(F[4]))
    plt.legend()
    plt.show()

def calculate_kelly(df, ret_f = 0.04 / 250):
    # 
    ret = df.diff(1)/df.shift(1)
    ret.ix[0,:] = 0
    # 
    C = np.mat(((ret - ret_f).cov() * 250).values)
    # 
    M = np.mat(((ret - ret_f).mean() * 250).values).T
    # Kelly
    F = C.I * M
    return F.T.tolist()[0]

def montecarlo_sim(df):
    # 
    ret = df.diff(1)/df.shift(1)
    ret.ix[0,:] = 0
    # 
    ret_f = 0.04/250
    # 
    ret_list = []
    s_list = []
    
    for i in range(10000):
        h = np.random.random(len(df.columns))
        h = h / np.sum(h)
        ret_list.append(np.sum((ret - ret_f).mean() * h) * 250)   
        s_list.append(np.sqrt(np.dot(np.dot(h.T, (ret - ret_f).cov() * 250), h)))   

    ret_array = np.array(ret_list)
    s_array = np.array(s_list)
    # 
    plt.figure()
    plt.scatter(x = s_array, y = ret_array, c = ret_array/s_array, marker='o')
    plt.grid()
    plt.xlabel('expected std')
    plt.ylabel('expected return')
    plt.colorbar(label='Sharpe Ratio')
    plt.show()

import scipy.optimize

def mvm(df,mod = 'max_sharpe'):
    # list(h),df
    def cal_stat(h):
        # 
        ret = df.diff(1)/df.shift(1)
        ret.ix[0,:] = 0
        # 
        ret_f = 0.04/250
        # 
        h = np.array(h).T
        # 
        ER = np.sum((ret - ret_f).mean() * h) * 250
        # 
        s = np.sqrt(np.dot(np.dot(h.T, (ret - ret_f).cov() * 250), h))
        return s, ER/s, ER/250 - 1 * s**2
    # 
    def min_sigma(h):
        return cal_stat(h)[0]
    # sharpe-1 * sharpe
    def max_sharpe(h):
        return -1 * cal_stat(h)[1]
    # 
    def max_return(h):
        return -1 * cal_stat(h)[2]
    
    # sum(h) - 1 = 0
    cons = ({'type':'eq', 'fun':lambda x: np.sum(x) - 1})
    # (0,1)
    bnds = tuple((0, 1) for x in range(len(df.columns)))
    # 
    initGuess = len(df.columns) * [1 / len(df.columns)]
    if mod == 'min_sigma':
        opts = scipy.optimize.minimize(min_sigma, initGuess, method='SLSQP', bounds=bnds, constraints=cons)
    elif mod == 'max_sharpe':
        opts = scipy.optimize.minimize(max_sharpe, initGuess, method='SLSQP', bounds=bnds, constraints=cons)
    elif mod == 'max_return':
        opts = scipy.optimize.minimize(max_return, initGuess, method='SLSQP', bounds=bnds, constraints=cons)
    return opts.x.tolist()
